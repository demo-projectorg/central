#!/usr/bin/env groovy
properties([
    [$class: 'GithubProjectProperty',
    displayName: '',
    projectUrlStr: 'https://github.com/harika-atmala/central.git/'],
    pipelineTriggers([githubPush()])])


//properties([pipelineTriggers([upstream('branch3')])])


            
pipeline {
    agent any
    
    parameters{
    string(defaultValue: 'default', description: 'Select which branch', name: 'Branch'), 
    choice(choices: ['Integration/unit'], description: 'Used for posgress.sh and CodeDeploy deployment group', name: 'Environment'), 
    booleanParam(defaultValue: false, description: 'Should we rebuild the database?', name: 'Rebuild Database'), 
    booleanParam(defaultValue: false, description: 'Deploy the application?', name: 'Deploy')
    }


    stages {
        stage('Buil') { 
            steps { 
                sh 'pwd' 
                echo ' new update...'
                echo 'name: ${params.Branch}'
                echo 'name: ${Environment}'
                echo 'name: {Rebuild Database}'
                echo 'name: {Deploy}'
            }
        }
        stage('Test'){
            steps {
                sh 'java -version'
                
            }
        }
        stage('Deploy') {
            steps {
                sh 'ls'
                sh 'pwd'
            }
        }
    }
}
